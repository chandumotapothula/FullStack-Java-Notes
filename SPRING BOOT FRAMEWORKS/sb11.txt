Spring Boot Thymeleaf 
======================
Thymeleaf is a Java template engine for rendering HTML (or XML) pages. It allows you to create web pages that can display dynamic data coming from your backend.Itâ€™s often used as a view layer in Spring MVC applications.

index.html
----------

	<head>
			<title>MyPage!</title>
			
			<!-- add external css -->
			<link rel="stylesheet" href="css/mystyles.css">
	</head>
	<body>
			<h1> Welcome to ThymeLeaf </h1>
			<p> Our First Application </p>
	</body>
</html>

package com.ihub.www.model;

public class User 
{
	private int id;
	private String email;
	private String role;
	
	public User(int id, String email, String role) {
		super();
		this.id = id;
		this.email = email;
		this.role = role;
	}

	public int getId() {
		return id;
	}

	public void setId(int id) {
		this.id = id;
	}

	public String getEmail() {
		return email;
	}

	public void setEmail(String email) {
		this.email = email;
	}

	public String getRole() {
		return role;
	}

	public void setRole(String role) {
		this.role = role;
	}
}

UserController.java
-------------------
package com.ihub.www.controller;

import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;

import com.ihub.www.model.User;

@Controller
public class UserController 
{
	@GetMapping("/index")
	public String home(Model model)
	{
		User u = new User(101,"raja@gmail.com","admin");
		model.addAttribute("user",u);
		return "index";
	}
}
